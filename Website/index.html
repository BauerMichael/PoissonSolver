Discretized Poisson Equation

Let $\Omega = (0,1)\times(0,1) \in \mathbb{R}^{2}$ a bounded, open intervall. We are searching for a function $u(x,y)$ which solves: \begin{eqnarray} -\Delta u(x,y) &=& f(x,y) \hspace{1mm} in \hspace{1mm} \Omega,\\ u(x,y) &=& g(x,y) \hspace{1mm} in \hspace{1mm} \partial \Omega. \end{eqnarray}

Jacobi (Relaxation) Method

For $\omega \in (0,1)$: \begin{equation} u^{k+1}_{i} = (1 - \omega)u^{k} + \frac {\omega} {a_{ii}} (f_{i} - \sum_{\substack{j = 1 \\ j \ne i}}^{n} a_{ij}u^{k}_{i}). \end{equation}

Conjugate Gradient Method

Let $A \in \mathbb{R}^{n}$ s.p.d. Choose an arbitrary $u^{0} \in \mathbb{R}^{n}$. Set $\beta_{-1} := 0$ and $r^{0} = b - Au^{0}$. \\ For $k = 1,2,...$, and if $r^{k-1} \ne 0$ calculate: \begin{eqnarray} &&p^{k-1} = r^{k-1} + \beta_{k-2}p^{k-2}, \hspace{1mm} \beta_{k-2} = \frac {\langle r^{k-1}, r^{k-1} \rangle} {\langle r^{k-2}, r^{k-2} \rangle}, \notag \\ &&u^{k} = u^{k-1} + \alpha_{k-1}p^{k-1}, \hspace{1mm} \alpha_{k-1} = \frac {\langle r^{k-1}, r^{k-1} \rangle} {\langle p^{k-1}, Ap^{k-1} \rangle}, \notag \\ &&r^{k} = r^{k-1} - \alpha_{k-1}Ap^{k-1}. \notag \end{eqnarray}

Preconditioned Conjugate Gradient Method

Let $A, W \in \mathbb{R}^{n}$ s.p.d. with an arbitrary initial guess $u^{0} \in \mathbb{R}^{n}$. Set $\beta_{-1} := 0$ and $r^{0} = b - Au^{0}$ and $z^{0} = W^{-1}r^{0}$ (solve $Wz^{0} = r^{0}$).\\ For $k = 1,2,...$ and if $r^{k-1} \ne 0$ calculate: \begin{eqnarray} &&p^{k-1} = z^{k-1} + \beta_{k-2}p^{k-2}, \hspace{1mm} \beta_{k-2} = \frac {\langle z^{k-1}, r^{k-1} \rangle} {\langle z^{k-2}, r^{k-2} \rangle}, \notag \\ &&u^{k} = u^{k-1} + \alpha_{k-1}p^{k-1}, \hspace{1mm} \alpha_{k-1} = \frac {\langle z^{k-1}, r^{k-1} \rangle} {\langle p^{k-1}, Ap^{k-1} \rangle}, \notag \\ &&r^{k} = r^{k-1} - \alpha_{k-1}Ap^{k-1}, \notag \\ &&z^{k} = W^{-1}r^{k}, \hspace{1mm} (solve Wz^{k} = r^{k}). \notag \end{eqnarray}

Incomplete LU decomposition

Let $A \in \mathbb{R}^{n \times n}$ and $E$ the pattern for $A$. Set $L = Id$, $R = 0$. Calculate for $i=1,2,...,n$: \begin{eqnarray} &&l_{i,i} = \left( a_{i,i} - \sum^{i-1}_{j=1,(i,j) \in E} l_{i,j}^{2} \right)^{\frac {1} {2}} \\ &&for \hspace{1mm} k = i+1,...,n: \hspace{1mm} if \hspace{1mm} (k,i) \in E: \\ &&l_{k,i} = \left( a_{k,i} - \sum^{k-1}_{j=1,(k,j) \in E, (i,j) \in E} l_{k,j} l_{i,j} \right) / l_{i,i} \end{eqnarray}

Modified incomplete LU decomposition

Let $A \in \mathbb{R}^{n \times n}$ s.p.d. and $E$ the pattern for $A$. Calculate for $i=1,2,...,n$: \begin{eqnarray} &&l_{i,i} = \left( a_{i,i} - \sum^{i-1}_{j=1,(i,j) \in E} l_{i,j}^{2} \right)^{\frac {1} {2}}. \\ &&for \hspace{1mm} k = i+1,...,n: \hspace{1mm} if \hspace{1mm} (k,i) \in E: \\ &&l_{k,i} = \left( a_{k,i} - \sum^{k-1}_{j=1,(k,j) \in E, (i,j) \in E} l_{k,j} l_{i,j} \right) / l_{i,i} \\ &&else:\\ &&a_{k,k} = a_{i,i} - \sum^{k-1}_{j=1,(k,j) \in E, (i,j) \in E} l_{k,j} l_{i,j} \end{eqnarray}

Multigrid V-Cycle

\begin{eqnarray} \textbf{V-cycle }(u,b) & \notag \\ \textbf{if }\textit{(coarsest grid) } &\hspace{2mm} return \hspace{1mm} u_{finest grid}=A^{-1}b \notag \\ \textbf{else } & \notag \\ \textit{a priori smooth } &\hspace{2mm} JacobiRelaxationMethod \notag \\ \textit{calculate residual } &r^{k} = b - Au^{k} \notag \\ \textit{restrict } &r^{k}_{2h}=Rr^{k}_{h} \notag \\ &A^{2h} = RA^{h}P \notag \\ \textit{recursion } &e^{k}_{2h} = \textbf{V-cycle}(0,r^{k}_{2h}) \notag \\ \textit{prolongate } &e^{k}_{h} = Pe^{k}_{2h} \notag \\ \textit{add error } &u^{k}_{h} = u^{k-1}_{h} + e^{k}_{h} \notag \\ \textit{a posteriori smooth } &\hspace{2mm} JacobiRelaxationMethod \notag \\ &\hspace{2mm} return \hspace{1mm} u_{h} \notag \\ \textbf{end} & \notag \end{eqnarray}

Multigrid W-Cycle

\begin{eqnarray} \textbf{W-cycle }(u,b) & \notag \\ \textbf{if }\textit{(coarsest grid) } &\hspace{2mm} return \hspace{1mm} u_{finest grid}=A^{-1}b \notag \\ \textbf{else } & \notag \\ \textit{a priori smooth } &\hspace{2mm} JacobiRelaxationMethod \notag \\ \textit{calculate residual } &r^{k} = b - Au^{k} \notag \\ \textit{restrict } &r^{k}_{2h}=Rr^{k}_{h} \notag \\ &A^{2h} = RA^{h}P \notag \\ \textit{recursion } &e^{k}_{2h} = \textbf{W-cycle}(0,r^{k}_{2h}) \notag \\ \textit{recursion } &e^{k}_{2h} = \textbf{W-cycle}(0,r^{k}_{2h}) \notag \\ \textit{prolongate } &e^{k}_{h} = Pe^{k}_{2h} \notag \\ \textit{add error } &u^{k}_{h} = u^{k-1}_{h} + e^{k}_{h} \notag \\ \textit{a posteriori smooth } &\hspace{2mm} JacobiRelaxationMethod \notag \\ &\hspace{2mm} return \hspace{1mm} u_{h} \notag \\ \textbf{end} & \notag \end{eqnarray}
